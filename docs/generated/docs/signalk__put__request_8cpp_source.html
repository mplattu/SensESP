10String SKRequest::send_request(DynamicJsonDocument& request,
11 std::function<void(DynamicJsonDocument&)> callback,
12 uint32_t timeout) {
13 // Create a new PendingRequest object to track this request...
14 PendingRequest* pending_request = new PendingRequest();
15
16 // Generate a uuid for the request...
17 pending_request->request_id = generate_uuid4();
18
19 // Save the callback for future processing...
20 pending_request->callback = callback;
21
22 // After 10 seconds, if we haven't already handled a response,
23 // assume its not coming.
24 pending_request->timeout_cleanup = app.onDelay(timeout, [pending_request]() {
25 // Mark the delay reaction null as it will be cleaned up by the ReactESP
26 // framework if this executes...
27 debugW("No response from server for request Id %s",
28 pending_request->request_id.c_str());
29 pending_request->timeout_cleanup = nullptr;
30 SKRequest::remove_request(pending_request->request_id);
31 });
32
33 request_map[pending_request->request_id] = pending_request;
34
35 // Now, send the actual request to the server...
36 request["requestId"] = pending_request->request_id;
37
38 String request_txt;
39 serializeJson(request, request_txt);
40 debugD("Sending websocket request to server: %s", request_txt.c_str());
41
42 sensesp_app->send_to_server(request_txt);
43
44 return pending_request->request_id;
45}

56void SKRequest::handle_response(DynamicJsonDocument& response) {
57 String request_id = response["requestId"];
58 PendingRequest* pending_request = get_request(request_id);
59 if (pending_request != nullptr) {
60 pending_request->callback(response);
61
62 // Now, are we done?
63 String state = response["state"];
64 if (!state.equalsIgnoreCase("PENDING")) {
65 remove_request(request_id);

67 } else {
68 debugW("Received request response for an untracked request: %s",

73void SKRequest::remove_request(String request_id) {
74 PendingRequest* pending_request = SKRequest::get_request(request_id);
75 if (pending_request != nullptr) {
76 // First, stop any pending timeout handlers...
77 if (pending_request->timeout_cleanup != nullptr) {
78 // The timeout code was not called, so just
79 // remove it from the ReactESP execution queue...
80 pending_request->timeout_cleanup->remove();
81 }
82
83 // Now, remove the request from the map...
84 request_map.erase(request_id);
85
86 // Finally, discard the request tracker...
87 delete pending_request;
88 }
