38// BME280 outputs temp in Celsius. Need to convert to Kelvin before sending to
39// Signal K. Pressure is output in Pascals, Humidity is output in relative
40// humidity (0 - 100%) and needs to be converted to ratio (0-1).Dew point
41// temperature is in kelvin and calculated from temperature and humidity.
42void BME280Value::enable() {
43 app.onRepeat(read_delay_, [this]() {
44 if (val_type_ == temperature) {
45 output = bme280_->adafruit_bme280_->readTemperature() +
46 273.15; // Kelvin is Celsius + 273.15
47 } else if (val_type_ == pressure) {
48 output = bme280_->adafruit_bme280_->readPressure();
49 } else if (val_type_ == humidity) {
50 output = bme280_->adafruit_bme280_->readHumidity() / 100;
51
52 } else {
53 output = 0.0;
54 }
55
56 notify();
57 });
58}
