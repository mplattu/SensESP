96 permission_strings[(int)requested_permissions_]);
97
98 // connect the system status controller
99
100 this->networking_->connect_to(&system_status_controller_);
101 this->ws_client_->connect_to(&system_status_controller_);
102
103 // create the wifi disconnect watchdog
104
105 this->system_status_controller_
106 .connect_to(new DebounceTemplate<SystemStatus>(
107 3*60*1000, // 180 s = 180000 ms = 3 minutes
108 "/system/wifi_reboot_watchdog"))
109 ->connect_to(new LambdaConsumer<SystemStatus>([](SystemStatus input) {
110 debugD("Got system status: %d", (int)input);
111 if (input == SystemStatus::kWifiDisconnected ||
112 input == SystemStatus::kWifiNoAP) {
113 debugW("Unable to connect to wifi for too long; restarting.");
114 app.onDelay(1000, []() { ESP.restart(); });
115 }
116 }));
117
118 // create a system status led and connect it
119
120 if (system_status_led_ == NULL) {
121 system_status_led_ = new SystemStatusLed(LED_PIN);
122 }
123 this->system_status_controller_.connect_to(system_status_led_);
