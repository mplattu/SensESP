193void WSClient::on_receive_put(DynamicJsonDocument& message) {
194 // Process PUT requests...
195 JsonArray puts = message["put"];
196 size_t responseCount = 0;
197 for (size_t i = 0; i < puts.size(); i++) {
198 JsonObject put = puts[i];
199 const char* path = put["path"];
200 String strVal = put["value"].as<String>();
201 debugD("Received PUT request for path %s (value %s)\n", path,
202 strVal.c_str());
203 const std::vector<SKPutListener*>& listeners =
204 SKPutListener::get_listeners();
205 for (size_t i = 0; i < listeners.size(); i++) {
206 SKPutListener* listener = listeners[i];
207 if (listener->get_sk_path().equals(path)) {
208 listener->parse_value(put);
209 responseCount++;
210 }
211 }
212
213 // Send back a request reasponse...
214 DynamicJsonDocument putResponse(512);
215 putResponse["requestId"] = message["requestId"];
216 if (responseCount == puts.size()) {
217 // We found a response for every PUT request
218 putResponse["state"] = "COMPLETED";
219 putResponse["statusCode"] = 200;
220 } else {
221 // One or more requests did not have a matching path
222 putResponse["state"] = "FAILED";
223 putResponse["statusCode"] = 405;
224 }
225 String responseTxt;
226 serializeJson(putResponse, responseTxt);
227 debugD("Replying to PUT request with %s", responseTxt.c_str());
228 this->client_.sendTXT(responseTxt);
229 }

253 debugI("WSClient websocket connect attempt (state=%d)",
254 connection_state_.get());
255
256 if (connection_state_ != WSConnectionState::kWSDisconnected) {
257 return;
258 }
259
260 if (!WiFi.isConnected()) {

262 "WiFi is disconnected. SignalK client connection will connect when "
263 "WiFi is connected.");
264 return;
265 }
266
267 debugD("Initiating websocket connection with server...");
268
269 connection_state_ = WSConnectionState::kWSAuthorizing;
270
271 String server_address = this->server_address_;
272 uint16_t server_port = this->server_port_;
273
274 if (this->server_address_.isEmpty()) {
275 if (!get_mdns_service(server_address, server_port)) {
276 debugE("No Signal K server found in network when using mDNS service!");
277 } else {
278 debugI("Signal K server has been found at address %s:%d by mDNS.",
279 server_address.c_str(), server_port);
280 }
281 }
282
283 if (!server_address.isEmpty() && server_port > 0) {
284 debugD("Websocket is connecting to Signal K server on address %s:%d",
285 server_address.c_str(), server_port);
286 } else {
287 // host and port not defined - wait for mDNS
288 connection_state_ = WSConnectionState::kWSDisconnected;
289 return;
290 }
291
292 if (this->polling_href_ != "") {
293 // existing pending request
294 this->poll_access_request(server_address, server_port, this->polling_href_);
295 return;
296 }
297
298 if (this->auth_token_ == NULL_AUTH_TOKEN) {
299 // initiate HTTP authentication
300 debugD("No prior authorization token present.");
301 this->send_access_request(server_address, server_port);
302 return;
303 }
304
305 if (test_auth_on_each_connect_ || !token_test_success_) {
306 // Test the validity of the authorization token for the first time...
307 this->test_token(server_address, server_port);
308 } else {
309 // The token has already been validated once,
310 // so we must be trying a subsequent reconnect.
311 // Jump directly to opening up the websocket...
312 server_detected_ = true;
313 this->connect_ws(server_address, server_port);
314 }

358void WSClient::send_access_request(const String server_address,
359 const uint16_t server_port) {
360 debugD("Preparing a new access request");
361 if (client_id_ == "") {
362 // generate a client ID
363 client_id_ = generate_uuid4();
364 save_configuration();
365 }
366
367 // create a new access request
368 DynamicJsonDocument doc(1024);
369 doc["clientId"] = client_id_;
370 doc["description"] = String("SensESP device: ") + sensesp_app->get_hostname();
371 doc["permissions"] = this->sk_permission_;
372 String json_req = "";
373 serializeJson(doc, json_req);
374
375 HTTPClient http;
376
377 String url = String("http://") + server_address + ":" + server_port +
378 "/signalk/v1/access/requests";
379 http.begin(wifi_client_, url);
380 http.addHeader("Content-Type", "application/json");
381 int httpCode = http.POST(json_req);
382 String payload = http.getString();
383 http.end();
384
385 // if we get a response we can't handle, try to reconnect later
386 if (httpCode != 202) {
387 debugW("Can't handle response %d to access request.", httpCode);
388 debugD("%s", payload.c_str());
389 connection_state_ = WSConnectionState::kWSDisconnected;
390 client_id_ = "";
391 return;
392 }
393
394 // http status code 202
395
396 deserializeJson(doc, payload.c_str());
397 String state = doc["state"];
398
399 if (state != "PENDING") {
400 debugW("Got unknown state: %s", state.c_str());
401 connection_state_ = WSConnectionState::kWSDisconnected;
402 client_id_ = "";
403 return;
404 }
405
406 String href = doc["href"];
407 polling_href_ = href;
408 save_configuration();
409
410 debugD("Polling %s in 5 seconds", polling_href_.c_str());
411 app.onDelay(5000, [this, server_address, server_port, href]() {
412 this->poll_access_request(server_address, server_port, this->polling_href_);
413 });

416void WSClient::poll_access_request(const String server_address,
417 const uint16_t server_port,
418 const String href) {
419 debugD("Polling SK Server for authentication token");
420
421 HTTPClient http;
422
423 String url = String("http://") + server_address + ":" + server_port + href;
424 http.begin(wifi_client_, url);
425 int httpCode = http.GET();
426 if (httpCode == 200 or httpCode == 202) {
427 String payload = http.getString();
428 http.end();
429 DynamicJsonDocument doc(1024);
430 auto error = deserializeJson(doc, payload.c_str());
431 if (error) {
432 debugW("WARNING: Could not deserialize http payload.");
433 debugW("DeserializationError: %s", error.c_str());
434 return; // TODO: return at this point, or keep going?
435 }
436 String state = doc["state"];
437 debugD("%s", state.c_str());
438 if (state == "PENDING") {
439 app.onDelay(5000, [this, server_address, server_port, href]() {
440 this->poll_access_request(server_address, server_port, href);
441 });
442 return;
443 } else if (state == "COMPLETED") {
444 JsonObject access_req = doc["accessRequest"];
445 String permission =
446 access_req["permission"]; // TODO: like this in ArdJson 6? String
447 // permission =
448 // resp["accessRequest"]["permission"];
449 polling_href_ = "";
450 save_configuration();
451
452 if (permission == "DENIED") {
453 debugW("Permission denied");
454 connection_state_ = WSConnectionState::kWSDisconnected;
455 return;
456 } else if (permission == "APPROVED") {
457 debugI("Permission granted");
458 String token = access_req["token"];
459 auth_token_ = token;
460 save_configuration();
461 app.onDelay(0, [this, server_address, server_port]() {
462 this->connect_ws(server_address, server_port);
463 });
464 return;
465 }
466 }
467 } else {
468 http.end();
469 if (httpCode == 500) {
470 // this is probably the server barfing due to
471 // us polling a non-existing request. Just
472 // delete the polling href.
473 debugD("Got 500, probably a non-existing request.");
474 polling_href_ = "";
475 save_configuration();
476 connection_state_ = WSConnectionState::kWSDisconnected;
477 return;
478 }
479 // any other HTTP status code
480 debugW("Can't handle response %d to pending access request.\n", httpCode);
481 connection_state_ = WSConnectionState::kWSDisconnected;
482 return;
483 }
