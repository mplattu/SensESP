29void ClickType::on_button_press() {
30 debugD(
31 "ClickType received PRESS on click count %d (millis: %ld, last release: "
32 "%ld ms ago)",
33 click_count_, millis(), (long)release_duration_);
34
35 if (click_count_ == 0) {
36 // This is a new, isolated "click" that we have not yet processed.
37 click_count_++;
38 press_duration_ = 0;
39 } else {
40 // One or more presses is already in progress...
41
42 if (press_duration_ > ultra_long_click_delay_) {
43 // The button down is the second one reported in a row without a button
44 // release and the press is now long enough to qualify as an "ultra long
45 // click"
46 on_ultra_long_click();
47 } else if (release_duration_ <= double_click_interval_) {
48 // This is the start of a second click to come in prior to the expiration
49 // of the double_click_interval. Remove any "SingleClick" report that may
50 // have been queued up....
51 if (delayed_click_report_ != NULL) {
52 delayed_click_report_->remove();
53 delayed_click_report_ = NULL;
54 debugD(
55 "ClickType press is double click. Removed queued SingleClick "
56 "report");
57 }
58 click_count_++;
59 }
60 }
61
62 this->emit(ClickTypes::ButtonPress);
63}

65void ClickType::on_button_release() {
66 debugD(
67 "ClickType received UNPRESS for click count %d (millis: %ld, press "
68 "duration: %ld ms)",
69 click_count_, millis(), (long)press_duration_);
70
71 if (click_count_ > 0) {
72 // This is the "release" of a click we are tracking...
73 if (press_duration_ >= this->ultra_long_click_delay_) {
74 on_ultra_long_click();
75 this->on_click_completed();
76 } else if (press_duration_ >= this->long_click_delay_) {
77 debugD(
78 "ClickType detected LongSingleClick (millis: %ld, press duration %ld "
79 "ms)",
80 millis(), (long)press_duration_);
81 this->emitDelayed(ClickTypes::LongSingleClick);
82 this->on_click_completed();
83 } else if (this->click_count_ > 1) {
84 // We have just ended a double click. Sent it immediately...
85 debugD(
86 "ClickType detected DoubleClick (millis: %ld, press duration %ld ms)",
87 millis(), (long)press_duration_);
88 this->emitDelayed(ClickTypes::DoubleClick);
89 this->on_click_completed();
90 } else {
91 // This is the end of a potential single click. Queue up the send of a
92 // SingleClick report, but delay it in case another click comes in prior
93 // to the double_click_interval, which would turn this click into a
94 // DoubleClick
95 unsigned long time_of_event = millis();
96 long pd = (long)press_duration_;
97 delayed_click_report_ =
98 app.onDelay(double_click_interval_ + 20, [this, pd, time_of_event]() {
99 debugD(
100 "ClickType detected SingleClick (millis: %ld, queue time: %ld, "
101 "press duration %ld ms)",
102 millis(), time_of_event, pd);
103 this->emit(ClickTypes::SingleClick);
104 this->on_click_completed();
105 });
106 }
107
108 release_duration_ = 0;
109 this->emit(ClickTypes::ButtonRelease);
110
111 } else {
112 // A press release with no initial press should happen only when
113 // an UltraLongClick has already been sent, or the producer
114 // is feeding us weird values...
115 release_duration_ = 0;
116 debugW("ClickType detected UNPRESS with no pending PRESS (millis=%ld)",
117 millis());
118 }
119}
