34void BaseBlinker::blip(int duration) {
35 // indicator for a blip being in progress
36 static bool blipping = false;
37
38 // only allow one blip at a time
39 if (blipping) {
40 return;
41 }
42 blipping = true;
43
44 bool orig_state = this->state_;
45 this->set_state(false);
46 int current_counter = this->update_counter_;
47 app.onDelay(duration, [this, duration, orig_state, current_counter]() {
48 // only update if no-one has touched the LED in the meanwhile
49 if (this->update_counter_ == current_counter) {
50 this->set_state(true);
51 int new_counter = this->update_counter_;
52 app.onDelay(duration, [this, orig_state, new_counter]() {
53 // again, only update if no-one has touched the LED
54 if (this->update_counter_ == new_counter) {
55 this->set_state(orig_state);
56 }
57 blipping = false;
58 });
59 } else {
60 blipping = false;
61 }
62 });
63}

125void PatternBlinker::tick() {
126 if (!enabled_) {
127 return;
128 }
129 // When pattern[pattern_ptr] == PATTERN_END, that's the end of the pattern,
130 // so start over at the beginning.
131 if (pattern_[pattern_ptr_] == PATTERN_END) {
132 pattern_ptr_ = 0;
133 }
134 // odd indices indicate times when LED should be OFF, even when ON
135 bool new_state = (pattern_ptr_ % 2) == 0;
136 this->set_state(new_state);
137 reaction_ = app.onDelay(pattern_[pattern_ptr_++], [this]() { this->tick(); });
138}
