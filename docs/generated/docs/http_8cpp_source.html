166 p->size());
167 } else if (p->isPost()) {
168 debugD("_POST[%s]: %s\n", p->name().c_str(), p->value().c_str());
169 } else {
170 debugD("_GET[%s]: %s\n", p->name().c_str(), p->value().c_str());
171 }
172 }

174 request->send(404);
